### 正文数据格式介绍

```json
/**
	content： 文件的地址
	contentType:文件类型
		html: html 后缀文件
		wword：doc docx wps 后缀文件
		wexecl: xls xlsx et 后缀文件
		dianjuPdf: pdf后缀文件
		dianjuOfd: ofd后缀文件
{
    // 原文件：文件类型可以为：html、wword、wexecl、dianjuPdf、dianjuOfd
    "initial": {
        "content": "3373936566841366719-/v8-seeyon-demo-private/cop/cip-capability",
        "contentType": "wword"
    },
    
    // 花脸数据：正文套红、清稿动作时产生 文件类型一定为wword
    "hualian": [
        {
            "type": "current", // 当前最新word文件
            "value": "0512147470824958537-/v8-seeyon-demo-private/cop/cip-capability",
            "time": 1716274114764
        },
        {
            "type": "original",  // 原正文花脸，第一次套红时保存的没有红头的原始文件
            "value": "3374200993729312161-/v8-seeyon-demo-private/cop/edoc335172694483814428",
            "time": 1716274109665,
            "position": 1
        }，
        {
            "type": "qghl",    // 清稿花脸，每一次清稿时保存的花脸，最多显示10个
            "value": "3374200993729312161-/v8-seeyon-demo-private/cop/edoc335172694483814428",
            "time": 1716274109665,
            "position": 2
```

```
/**
    * @description: 获取盖章最新文件地址
    * @Date: 2026/3/17
    */
public String getFileSK(JSONObject contentJson) {
String storageKey = "";
//获取盖章最新文件
if (contentJson.containsKey("gaizhang") && contentJson.getJSONObject("gaizhang") != null) {
    JSONObject obj = contentJson.getJSONObject("gaizhang");
    storageKey = obj.getStr("content");
} else if (contentJson.containsKey("edition") && contentJson.getJSONObject("edition") != null) {
    JSONObject obj = contentJson.getJSONObject("edition");
    storageKey = obj.getStr("content");
} else if (contentJson.containsKey("taohong") && contentJson.getJSONObject("taohong") != null) {
    JSONObject obj = contentJson.getJSONObject("taohong");
    storageKey = obj.getStr("content");
} else if (contentJson.containsKey("compose") && contentJson.getJSONObject("compose") != null) {
    JSONObject obj = contentJson.getJSONObject("compose");
    storageKey = obj.getStr("content");
} else if (contentJson.containsKey("initial") && contentJson.getJSONObject("initial") != null) {
    JSONObject obj = contentJson.getJSONObject("initial");
    if (obj.containsKey("content") && obj.getStr("content") != null) {
        storageKey = obj.getStr("content");
    }
}
return storageKey;
}
```
